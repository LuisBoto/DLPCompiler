struct Piloto {
	inicial:char;  
	crono:[3]float;	// Tiempos en la Q1, Q2 y Q3
};

var F1:[3] Piloto;

inicia(i:int, inicial:char, q1:float, q2:float, q3:float) {
	F1[i].inicial = inicial;
	F1[i].crono[0] = q1;
	F1[i].crono[1] = q2;
	F1[i].crono[2] = q3;
}
